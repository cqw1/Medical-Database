 19 /**
 20  * Represents a set of contraints applied to an index or a store.
 21  * The way this object is designed, you only have to set the properties that are meaningful for your
 22  * requirement. Not setting a property means that any value is ok.
 23  * @constructor
 24  */
 25 fullproof.Capabilities = function() {
 26 	if (!(this instanceof fullproof.Capabilities)) {
 27 		return new fullproof.Capabilities();
 28 	}
 29 };
 30 
 31 /**
 32  * Compares a value with a property of this Capabilities object.
 33  * @param property a property name
 34  * @param value the valued to be compared
 35  * @protected
 36  * This should probably be made private or something
 37  */
 38 fullproof.Capabilities.prototype.matchValue = function (property, value) {
 39     if (value === undefined) {
 40         return true;
 41     } else if (typeof property == "object" && property.constructor == Array) {
 42         for (var i = 0; i < property.length; ++i) {
 43             if (property[i] === value) {
 44                 return true;
 45             }
 46         }
 47         return false;
 48     } else {
 49         return property === value;
 50     }
 51 };

144 /**
145  * Returns true if the current Capabilities object subsumes another Capabilities.
146  * @param otherCapabilities the Capabilities that must be subsumed by the current instance
147  */
148 fullproof.Capabilities.prototype.isCompatibleWith = function (otherCapabilities) {
149     var objstore = this.matchValue(this.canStoreObjects, otherCapabilities.canStoreObjects);
150     var isvol = this.matchValue(this.isVolatile, otherCapabilities.isVolatile);
151     var score = this.matchValue(this.useScores, otherCapabilities.useScores);
152     var isavail = this.isAvailable === true;
153 
154     return objstore && isvol && isavail && score;
155 };
