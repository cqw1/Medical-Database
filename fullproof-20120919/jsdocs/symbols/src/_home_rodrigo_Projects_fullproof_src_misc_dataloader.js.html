 22 	function getNewXmlHttpRequest()  {
 23 		if (typeof window.XMLHttpRequest !== "undefined") {
 24 			return new window.XMLHttpRequest;
 25 		} else {
 26 			
 27 		} if (typeof window.ActiveXObject === 'function') {
 28 		        try { return new ActiveXObject('Msxml2.XMLHTTP.6.0'); } catch(e) {}
 29 		        try { return new ActiveXObject('Msxml2.XMLHTTP.3.0'); } catch(e) {}
 30 		        try { return new ActiveXObject('Microsoft.XMLHTTP'); } catch(e) {}
 31 			}
 32 		throw "Error, can't find a suitable XMLHttpRequest object";
 33 	}

 35 	fullproof.DataLoader = function () {
 36 
 37         if (!(this instanceof fullproof.DataLoader)) {
 38             return new fullproof.DataLoader();
 39         }
 40 
 41         var loadQueue = [];
 42         var currentQueue = [];
 43 
 44         this.setQueue = function () {
 45             for (var i = 0; i < arguments.length; ++i) {
 46                 if (arguments[i].constructor == Array) {
 47                     loadQueue = loadQueue.concat(arguments[i]);
 48                 } else {
 49                     loadQueue.push(arguments[i]);
 50                 }
 51             }
 52             return this;
 53         };
 54 
 55         var processQueue = function (completeCallback, fileLoadedCallback, fileFailedCallback) {
 56 
 57             if (currentQueue.length == 0) {
 58                 completeCallback();
 59                 return;
 60             }
 61 
 62             var element = currentQueue.shift();
 63 
 64             var request = getNewXmlHttpRequest();
 65             request.onreadystatechange = function () {
 66                 if (request.readyState == 4) {
 67                     if (request.status != 200) {
 68                         // Handle error, e.g. Display error message on page
 69                         if (fileFailedCallback) {
 70                             fileFailedCallback(element);
 71                             processQueue(completeCallback, fileLoadedCallback, fileFailedCallback);
 72                         }
 73                     } else {
 74                         var serverResponse = request.responseText;
 75                         if (fileLoadedCallback) {
 76                             fileLoadedCallback(serverResponse, element);
 77                             processQueue(completeCallback, fileLoadedCallback, fileFailedCallback);
 78                         }
 79                     }
 80                 }
 81             };
 82             request.open("GET", element, true);
 83             request.send(null);
 84         };
 85 
 86         this.start = function (completeCallback, fileLoadedCallback, fileFailedCallback) {
 87             currentQueue = [].concat(loadQueue);
 88             processQueue(completeCallback, fileLoadedCallback, fileFailedCallback);
 89         };
 90     };
