 19 /**
 20  * A TextInjector associates an index and an analyzer to provide an
 21  * object able to inject texts.
 22  * 
 23  * @constructor
 24  * @param index the index to use when injecting
 25  * @param analyzer the analyzer to use to parse and normalize the text
 26  * TODO: move this object in its own place or in utils
 27  */
 28 fullproof.TextInjector = function(index, analyzer) {
 29 	if (!(this instanceof fullproof.TextInjector)) {
 30 		return new fullproof.TextInjector(index,analyzer);
 31 	}
 32 	this.index = index;
 33 	this.analyzer = analyzer;
 34 };

 36 /**
 37  * Inject a text and associates each of the word it composed with the provided value.
 38  * @param text some text to inject in the index
 39  * @param value the value associated to each word from the text
 40  * @param callback a function to call when the injection is complete
 41  */
 42 fullproof.TextInjector.prototype.inject = function(text,value,callback) {
 43 	var self = this;
 44 	this.analyzer.getArray(text, function(array_of_words) {
 45 		var synchro = fullproof.make_synchro_point(callback, array_of_words.length);
 46 		for (var i=0; i<array_of_words.length; ++i) {
 47 			var val = array_of_words[i];
 48 			if (val instanceof fullproof.ScoredEntry) {
 49 				val.value = val.value===undefined?value:val.value;
 50 				self.index.inject(val.key, val, synchro); // the line number is the value stored
 51 			} else {
 52 				self.index.inject(array_of_words[i], value, synchro); // the line number is the value stored
 53 			}
 54 		}
 55 	});
 56 };

 58 /**
 59  * Bulk-inject an array of  text and an array of values. The text injected is associated to the value
 60  * of the same index from the valueArray array. An optional progress function is called at regular interval
 61  * to provide a way for the caller to give user feedback on the process.
 62  * 
 63  * @param texArray an array of text
 64  * @param valueArray an array of values. The length of valueArray must equal the length of textArray.
 65  * @param callback a function to call when the injection is complete
 66  * @param progressCallback a function called with progress indication. A numeric argument is provided to 
 67  * the function, which ranges from 0 to 1 (0 meaning not done, 1 meaning complete). Note that due to the
 68  * floating nature of numeric values in javascript, you should not rely on receiving a 1, rather use
 69  * the callback function parameter, which will be always called on completion of the injection.
 70  */
 71 fullproof.TextInjector.prototype.injectBulk = function(textArray, valueArray, callback, progressCallback) {
 72 	var words = [];
 73 	var values = [];
 74 	var self = this;
 75 	for (var i=0, max=Math.min(textArray.length, valueArray.length); i<max; ++i) {
 76 		(function(text,value) {
 77 			self.analyzer.getArray(text, function(array_of_words) {
 78 				for (var w=0; w<array_of_words.length; ++w) {
 79 					var val = array_of_words[w];
 80 					if (val instanceof fullproof.ScoredEntry) {
 81 						val.value = val.value===undefined?value:val.value;
 82 						words.push(val.key);
 83 						values.push(val);
 84 					} else {
 85 						words.push(val);
 86 						values.push(value);
 87 					}
 88 				}
 89 			});
 90 		})(textArray[i], valueArray[i]);
 91 	}
 92 	this.index.injectBulk(words,values, callback, progressCallback);
 93 };

 95 /**
 96  * Represent all the data associated to an index, from the point of view of a search engine.
 97  * @param name the name of the index, expected to be unique in the search engine
 98  * @param capabilities a fullproof.Capabilities
 99  * @param analyzer
100  * @param initializer
101  * @param index
102  * @constructor
103  */
104 fullproof.IndexUnit = function (name, capabilities, analyzer, initializer, index) {
105     /**
106      * The name of the index
107      */
108     this.name = name;
109     /**
110      * The fullproof.Capabilities object originally associated to the index
111      */
112     this.capabilities = capabilities;
113     /**
114      * The parser used to inject text in the index
115      */
116     this.analyzer = analyzer;
117     /**
118      * The initializer function when the index needs to be built
119      * @type {*}
120      */
121     this.initializer = initializer;
122     /**
123      * The index itself
124      */
125     this.index = index;
126 };

134 /**
135  * Adds an array of index units
136  * @param indexes an array of fullproof.IndexUnit instances
137  * @param callback the function to call when all the indexes are added
138  * @private
139  * @static
140  */
141 fullproof.AbstractEngine.addIndexes = function (engine, indexes, callback) {
142     var starter = false;
143     while (indexes.length > 0) {
144         var data = indexes.pop();
145         starter = (function (next, data) {
146             return function () {
147                 fullproof.AbstractEngine.addIndex(engine, data.name, data.analyzer, data.capabilities, data.initializer, next !== false ? next : callback);
148             };
149         })(starter, data);
150     }
151     if (starter !== false) {
152         starter();
153     }
154     return this;
155 };

157 /**
158  * Adds un index to the engine. It is not possible to add an index after the engine was opened.
159  * @param name the name of the engine
160  * @param the analyzer used to parse the text
161  * @param capabilities a fullproof.Capabilities instance describing the requirements for the index
162  * @param initializer a function called when the index is created. This function can be used to populate the index.
163  * @param completionCallback a function on completion, with true if the index was successfully added, false otherwise.
164  * @return this instance
165  * @private
166  * @static
167  */
168 fullproof.AbstractEngine.addIndex = function(engine, name, analyzer, capabilities, initializer, completionCallback) {
169 	var self = engine;
170 	var indexData = new fullproof.IndexUnit(name,capabilities,analyzer); 
171 
172 	if (!engine.checkCapabilities(capabilities, analyzer)) {
173 		return completionCallback(false);
174 	}
175 
176 	var indexRequest = new fullproof.IndexRequest(name, capabilities, function(index, callback) {
177 		var injector = new fullproof.TextInjector(index, indexData.analyzer);
178 		initializer(injector, callback);
179 	});
180 	
181 	if (engine.storeManager.addIndex(indexRequest)) {
182 		if (engine.indexes === undefined) {
183             engine.indexes = [];
184 		}
185         engine.indexes.push(indexData);
186         engine.indexesByName[name] = indexData;
187 		if (completionCallback) {
188 			completionCallback(true);
189 		}
190 		return true;
191 	} else {
192 		if (completionCallback) {
193 			completionCallback(false);
194 		}
195 		return false;
196 	}
197 };

199 /**
200  * Opens the engine: this function opens all the indexes at once, makes the initialization if needed,
201  *  and makes this engine ready for use. Do not use any function of an engine, except addIndex, before
202  *  having opened it.
203  *  @param indexArray an array of index descriptors. Each descriptor is an object that defines the name, analyzer, capabilities, and initializer properties.
204  *  @param callback function called when the engine is properly opened
205  *  @param errorCallback function called if for some reason the engine cannot open some index
206  */
207 fullproof.AbstractEngine.prototype.open = function (indexArray, callback, errorCallback) {
208     var self = this;
209     indexArray = (indexArray.constructor !== Array)?[indexArray]:indexArray; // Makes it an Array if it's not
210     fullproof.AbstractEngine.addIndexes(self, indexArray);
211 
212     this.storeManager.openIndexes(function (storesArray) {
213         self.storeManager.forEach(function (name, index) {
214             self.indexesByName[name].index = index;
215         });
216         callback(self);
217     }, errorCallback);
218     return this;
219 };

221 /**
222  * Inject a text document into all the indexes managed by the engine.
223  * @param text some text to be parsed and indexed
224  * @param value the primary value (number or string) associated to this object.
225  * @param callback the function called when the text injection is done
226  */
227 fullproof.AbstractEngine.prototype.injectDocument = function(text, value, callback) {
228 	var synchro = fullproof.make_synchro_point(function(data) {
229 		callback();
230 	});
231 
232 	this.forEach(function(name, index, parser) {
233 		if (name) {
234 			parser.parse(text, function(word) {
235 				if (word) {
236 					index.inject(word, value, synchro); // the line number is the value stored
237 				} else {
238 					synchro(false);
239 				}
240 			})
241 		}
242 	}, false);
243 	return this;
244 };

246 /**
247  * Clears all the indexes managed by this engine. Do not call this function
248  * before the engine was open()'ed.
249  * @param callback a function called when all the indexes are cleared.
250  */
251 fullproof.AbstractEngine.prototype.clear = function(callback) {
252     "use strict";
253     if (this.getIndexCount() === 0) {
254         return callback();
255     }
256 	var synchro = fullproof.make_synchro_point(callback, this.getIndexCount());
257 	this.forEach(function(name, index, parser) {
258 		if (name) {
259 			index.clear(synchro);
260 		} else {
261 			synchro(false);
262 		}
263 	});
264 };

285 /**

295 /**
296  * Iterates over the indexes, in order, and calls the callback function with 3 parameters:
297  * the name of the index, the index instance itself, and the analyzer associated to this index.
298  * @param callback the callback function(name,index,analyzer){}
299  * @return this engine instance
300  */
301 fullproof.AbstractEngine.prototype.forEach = function (callback) {
302     for (var i = 0, max = this.indexes.length; i < max; ++i) {
303         callback(this.indexes[i].name, this.indexes[i].index, this.indexes[i].analyzer);
304     }
305     for (var i = 1; i < arguments.length; ++i) {
306         callback(arguments[i]);
307     }
308     return this;
309 };
310 
311 fullproof.AbstractEngine.prototype.getIndexCount = function () {
312     return this.indexes.length;
313 };
