  9 var ScoreTestAnalyzer = function() {
 10     // Stores the normalizers... (don't store arguments, as it contains more than an array)
 11     var analyzer = new fullproof.StandardAnalyzer([]);
 12     this.provideScore = true;
 13 
 14     this.parse = function (text, callback) {
 15         var words = {};
 16         var wordcount = 1;
 17         var self = this;
 18         analyzer.parse(text, function (word) {
 19             if (word !== false) {
 20                 if (words[word] === undefined || words[word].constructor !== Array) {
 21                     words[word] = [];
 22                 }
 23                 words[word].push(wordcount++);
 24             } else {
 25                 // Evaluate the score for each word
 26                 for (var w in words) {
 27                     var res = words[w];
 28                     var score = 0;
 29                     for (var i = 0; i < res.length; ++i) {
 30                         score += res[i];
 31                     }
 32                     callback(new fullproof.ScoredEntry(w, undefined, score));
 33                 }
 34                 callback(false);
 35             }
 36         });
 37     };
 38 }
 39 ScoreTestAnalyzer.prototype = new fullproof.AbstractAnalyzer();

 41 function initializerFunc(injector, callback) {

 43 	for (var k in corpus) {

 45 	}

 48 function init_engine(indexes, callback) {
 49     var engine = new fullproof.ScoringEngine([new fullproof.StoreDescriptor("memorystore", fullproof.store.MemoryStore)]);
 50     engine.open(indexes, function() {
 51        callback(engine);
 52     });
