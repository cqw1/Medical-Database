 16 var fullproof = fullproof||{};
 17 (function() {
 18 "use strict";
 19 
 20     /**
 21      * A prototype for Analyzers objects.
 22      * @constructor
 23      */
 24     fullproof.AbstractAnalyzer = function() {
 25         /**
 26          * Sometimes it's convenient to receive the whole set of words cut and normalized by the
 27          * analyzer. This method calls the callback parameter only once, with as single parameter
 28          * an array of normalized words.
 29          * @param text some text to analyze
 30          * @param callback a function called with an array (possibly empty) of string
 31          */
 32         this.getArray = function(text, callback) {
 33             var parser_synchro = fullproof.make_synchro_point(function(array_of_words) {
 34                 callback(array_of_words);
 35             });
 36             this.parse(text, parser_synchro);
 37         };
 38 
 39     };
 40 
 41 	/**
 42 	 * A simple private parser that relies on the unicode letter/number
 43 	 * categories. Word boundaries are whatever is not a letter 
 44 	 * or a number.
 45 	 */
 46 	var simple_parser = function(str, callback, functor) {
 47 		functor = functor||net.kornr.unicode.is_letter_number;
 48 		var current_word = "";
 49 		for (var i=0,max=str.length; i<max; ++i) {
 50 			if (functor(str.charCodeAt(i))) {
 51 				current_word += str[i];
 52 			} else {
 53 				if (current_word.length>0) {
 54 					callback(current_word);
 55 					current_word = "";
 56 				}
 57 			}
 58 		}
 59 		if (current_word.length>0) {
 60 			callback(current_word);
 61 		}
 62 		callback(false);
 63 	};
 64 	
 65 	function arguments_to_array(args) {
 66 		var result = [];
 67 		for (var i=0; i<args.length; ++i) {
 68 			if (args[i].constructor == Array) {
 69 				result = result.concat(args[i]);
 70 			} else {
 71 				result.push(args[i]);
 72 			}
 73 		}
 74 		return result;
 75 	}
 76 	
 77 	/**
 78 	 * An analyzer with a parse() method. An analyzer does more than
 79 	 * just parse, as it normalizes each word calling the sequence
 80 	 * of normalizers specified when calling the constructor.
 81 	 * 
 82 	 * @constructor
 83 	 * @param normalizers... the constructor can take normalizers as parameters. Each 
 84 	 * normalizer is applied sequentially in the same order as they are
 85 	 * passed in the constructor.
 86 	 */
 87 	fullproof.StandardAnalyzer = function() {
 88         var normalizers = arguments_to_array(arguments);
 89 
 90         // Enforces new object
 91 		if (!(this instanceof fullproof.StandardAnalyzer)) {
 92 			return new fullproof.StandardAnalyzer(normalizers);
 93 		}
 94 
 95 		// Stores the normalizers... (don't store arguments, as it contains more than an array) 
 96 		this.provideScore = false;
 97 		/**
 98 		 * When true, the parser calls its callback function with 
 99 		 * the parameter {boolean}false when complete. This allows
100 		 * the callback to know when the parsing is complete. When
101 		 * this property is set to false, the parser never triggers
102 		 * the last call to callback(false).
103 		 * 
104 		 * @expose
105 		 */
106 		this.sendFalseWhenComplete = true;
107 		
108 		/**
109          * The main method: cuts the text in words, calls the normalizers on each word,
110          * then calls the callback with each non empty word.
111          * @param text the text to analyze
112          * @param callback a function called with each word found in the text.
113          */
114         this.parse = function (text, callback) {
115             var self = this;
116             simple_parser(text, function (word) {
117                 if (typeof word == "string") {
118                     word = word.trim();
119                     if (word != "") {
120                         for (var i = 0; i < normalizers.length; ++i) {
121                             word = normalizers[i](word);
122                         }
123                     }
124 
125                     if (callback && word && word != "") {
126                         callback(word);
127                     }
128                 } else if (word === false && self.sendFalseWhenComplete && callback) {
129                     callback(word);
130                 }
131             });
132         };
133 	};
134 
135     fullproof.StandardAnalyzer.prototype = new fullproof.AbstractAnalyzer();

137     /**
138      * The ScoringAnalyzer is not unlike the StandardAnalyzer, except that is attaches a score to each token,
139      * related to its place in the text. This is a very naive implementation, and therefore the adjustement
140      * is tweaked to be very light: it simplistically says that the more a token is near the start of the text,
141      * the more relevant it is to the document. Although very simple, it follows the normally expected form
142      * of a text where the headers and titles come first, and should provide decent result. You can use
143      * this as a basis and make a ScoringAnalyzer adapted to your data.
144      * @constructor
145      */
146 	fullproof.ScoringAnalyzer = function() {
147 		// Stores the normalizers... (don't store arguments, as it contains more than an array) 
148 		var normalizers = arguments_to_array(arguments);
149 		var analyzer = new fullproof.StandardAnalyzer(normalizers);
150 		this.sendFalseWhenComplete = analyzer.sendFalseWhenComplete = true;
151 		this.provideScore = true;
152 		
153 		this.parse = function (text, callback) {
154             var words = {};
155             var wordcount = 0;
156             var totalwc = 0;
157             var self = this;
158             analyzer.parse(text, function (word) {
159                 if (word !== false) {
160                     if (words[word] === undefined || words[word].constructor !== Array) {
161                         words[word] = [];
162                     }
163                     words[word].push(wordcount);
164                     totalwc += ++wordcount;
165                 } else {
166                     // Evaluate the score for each word
167                     for (var w in words) {
168                         var res = words[w];
169                         var offsetcount = 1;
170                         var occboost = 0;
171                         for (var i = 0; i < res.length; ++i) {
172                             occboost += (3.1415 - Math.log(1 + res[i])) / 10;
173                         }
174                         var countboost = Math.abs(Math.log(1 + res.length)) / 10;
175                         var score = 1 + occboost * 1.5 + countboost * 3;
176                         // console.log(w + ": " + words[w].join(",") + ", countboost: " + countboost + ", occboost: " + occboost);
177                         callback(new fullproof.ScoredEntry(w, undefined, score));
178                     }
179 
180                     if (self.sendFalseWhenComplete == true) {
181                         callback(false);
182                     }
183 
184                 }
185             });
186         };
187 		
188 
189 	};
190     fullproof.ScoringAnalyzer.prototype = new fullproof.AbstractAnalyzer();
191 
192 
193 })();
