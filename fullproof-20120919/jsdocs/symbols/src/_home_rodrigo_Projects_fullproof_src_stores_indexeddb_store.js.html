 35 	function install_on_request(req, success, error) {
 36 		req.onsuccess = success;
 37 		req.onerror = error;
 38 		return req;
 39 	}

 42 		if (tx.db.objectStoreNames.contains(name)) {
 43 			return tx.objectStore(name);
 44 		} else {
 45 			return tx.db.createObjectStore(name, parameter);
 46 		}

 49 	function setObject(store, object, callback, error) {
 50 		var req = store.put(object);
 51 		install_on_request(req, fullproof.make_callback(callback, object), error);
 52 	}

 54 	function getOrCreateObject(store, keyValue, defaultValue, callback, error) {
 55 
 56 		function create() {
 57 			if (fullproof.isFunction(defaultValue)) {
 58 				defaultValue = defaultValue();
 59 			}
 60 			setObject(store, defaultValue, callback, error);
 61 		}
 62 		try {
 63 			var req = store.get(keyValue);
 64 		} catch (e) {
 65 			console && console.log && console.log(e);
 66 		}
 67 
 68 		req.onsuccess = function(ev) {
 69 			if (ev.target.result === undefined) {
 70 				if (defaultValue !== undefined) {
 71 					create();
 72 				} else {
 73 					error();
 74 				}
 75 			} else {
 76 				callback(req.result);
 77 			}
 78 		};
 79 		req.onerror = create;
 80 	}

 83      * An IndexedDBIndex object manages an inverted index in an IndexedDB store.

 85      * @param database the database associated to this index
 86      * @param storeName the name of the object store

 89 	function IndexedDBIndex(parent, database, indexName, comparatorObject, useScore) {
 90 		this.parent = parent;
 91 		this.database = database;
 92 		this.name = indexName;
 93 		this.comparatorObject = comparatorObject;
 94 		this.useScore = useScore;
 95 		this.internalComparator = useScore?function(a,b) {
 96 			return this.comparatorObject(a.value,b.value);
 97 		}:function(a,b) {
 98 			return this.comparatorObject(a,b);
 99 		};
100 	}

102 	IndexedDBIndex.prototype.clear = function (callback) {
103         callback = callback || function(){};
104         var self = this;
105         var wrongfunc = fullproof.make_callback(callback, false);
106         var tx = this.database.transaction([this.name, this.parent.metaStoreName], fullproof.store.READWRITEMODE);
107         var metastore = tx.objectStore(this.parent.metaStoreName);
108         install_on_request(metastore.put({id:this.name, init:false}), function () {
109             fullproof.call_new_thread(function () {
110                 var tx = self.database.transaction([self.name], fullproof.store.READWRITEMODE);
111                 var store = tx.objectStore(self.name);
112                 var req = store.clear();
113                 install_on_request(req, fullproof.make_callback(callback, true), wrongfunc);
114             });
115         }, wrongfunc);
116     };

118 	IndexedDBIndex.prototype.inject = function(word, value, callback) {
119 		var tx = this.database.transaction([this.name], fullproof.store.READWRITEMODE);
120 		var store = tx.objectStore(this.name);
121 		var self = this;
122 		getOrCreateObject(store, word, function() { return {key:word,v:[]} }, function(obj) {
123 			var rs = new fullproof.ResultSet(self.comparatorObject).setDataUnsafe(obj.v);
124 			if (value instanceof fullproof.ScoredElement) {
125 				if (self.useScore) {
126 					rs.insert({v:value.value, s:value.score});
127 				} else {
128 					rs.insert(value.value);
129 				}
130 			} else {
131 				rs.insert(value);
132 			}
133 			obj.v = rs.getDataUnsafe();
134 			setObject(store, obj, callback, fullproof.make_callback(callback, false));
135 		}, fullproof.make_callback(callback,false));
136 	};

138     var storedObjectComparator_score = {
139         lower_than: function(a, b) {
140             return (a.v?a.v:a)<(b.v? b.v:b);
141         },
142         equals: function(a,b) {
143             return (a.v?a.v:a)===(b.v? b.v:b);
144         }

147     function createMapOfWordsToResultSet(self, wordArray, valuesArray, offset, count, resultPropertiesAsArray) {
148 		var result = new fullproof.HMap();
149 		for (; offset < count; ++offset) {
150 			var word = wordArray[offset];
151 			var value = valuesArray[offset];
152 
153 			if (result.get(word) === undefined) {
154 				result.put(word, new fullproof.ResultSet(storedObjectComparator_score));
155                 resultPropertiesAsArray.push(word);
156             }
157 			if (value instanceof fullproof.ScoredElement) {
158 				if (self.useScore) {
159                     var rs = result.get(word);
160                     rs.insert({v:value.value, s:value.score});
161 				} else {
162                     result.get(word).insert(value.value);
163 				}
164 			} else {
165                 result.get(word).insert(value);
166 			}
167 		}
168 		return result;
169 	}

171 	function storeMapOfWords(self, store, words, data, callback, offset, max) {
172 		if (words.length>0 && offset < max) {
173 			var word = words[offset];
174 			var value = data.get(word);
175 			getOrCreateObject(store, word, function() { return {key:word,v:[]};}, function(obj) {
176 				var rs = new fullproof.ResultSet(self.comparatorObject).setDataUnsafe(obj.v);
177 				rs.merge(value);
178 				obj.v = rs.getDataUnsafe();
179 				setObject(store, obj, function() {
180 					storeMapOfWords(self, store, words, data, callback, offset+1, max);
181 				}, function() { callback(false); });
182 			});
183 		} else {
184 			callback(true);
185 		}
186 	}

188 	IndexedDBIndex.prototype.injectBulk = function (wordArray, valuesArray, callback, progress) {
189         var self = this;
190         if (wordArray.length !== valuesArray.length) {
191             throw "Can't injectBulk, arrays length mismatch";
192         }
193 
194         var batchSize = 100;
195 
196         var words = [];
197         var data = createMapOfWordsToResultSet(this, wordArray, valuesArray, 0, wordArray.length, words);
198 
199         function storeData(self, words, data, callback, progress, offset) {
200             if (progress) {
201                 progress(offset / words.length);
202             }
203             var tx = self.database.transaction([self.name], fullproof.store.READWRITEMODE);
204             var store = tx.objectStore(self.name);
205             storeMapOfWords(self, store, words, data, function (res) {
206                 if (offset < words.length) {
207                     fullproof.call_new_thread(storeData, self, words, data, callback, progress, offset + batchSize)
208                 } else {
209                     callback(res);
210                 }
211             }, offset, Math.min(offset + batchSize, words.length));
212         }
213 
214         if (words.length > 0) {
215             storeData(this, words, data, callback, progress, 0)
216         } else {
217             callback(true);
218         }
219     };

221 	IndexedDBIndex.prototype.lookup = function(word, callback) {
222 		var tx = this.database.transaction([this.name]);
223 		var store = tx.objectStore(this.name);
224 		var self = this;
225 		getOrCreateObject(store, word, undefined, function(obj) {
226 			if (obj && obj.v) {
227 				var rs = new fullproof.ResultSet(self.comparatorObject);
228 				for (var i=0,max=obj.v.length; i<max; ++i) {
229 					var o = obj.v[i];
230 					if (self.useScore) {
231 						rs.insert(new fullproof.ScoredEntry(word, o.v, o.s));
232 					} else {
233 						rs.insert(o);
234 					}
235 				}
236 				callback(rs);
237 			} else {
238 				callback(new fullproof.ResultSet(self.comparatorObject));
239 			}
240 		}, function() { callback(new fullproof.ResultSet(self.comparatorObject)); });
241 	};

276 	function createStores(database, indexRequestArray, metaStoreName) {
277 		if (!database.objectStoreNames.contains(metaStoreName)) {
278 			database.createObjectStore(metaStoreName, {keyPath: "id"});
279 		}
280 		for (var i=0; i<indexRequestArray.length; ++i) {
281 			if (!database.objectStoreNames.contains(indexRequestArray[i].name)) {
282 				database.createObjectStore(indexRequestArray[i].name, {keyPath: "key"});
283 			}
284 		}

301 			for (var i=0; i<self.indexRequests.length; ++i) {
302 				var ireq =self.indexRequests[i];
303 				var compObj = ireq.capabilities.getComparatorObject()?ireq.capabilities.getComparatorObject():(self.useScore?fullproof.ScoredElement.comparatorObject:undefined);

310 		function callInitializerIfNeeded(database, self, indexRequestArray, callback, errorCallback) {
311 			if (indexRequestArray.length == 0) {
312 				return callback(true);
313 			}
314 			
315 			var tx = database.transaction([self.metaStoreName], fullproof.store.READWRITEMODE);
316 			var metastore = tx.objectStore(self.metaStoreName);
317 			var ireq = indexRequestArray.shift();
318             getOrCreateObject(metastore, ireq.name, {id: ireq.name, init: false},
319 					function(obj) {
320 						if (obj.init == false && ireq.initializer) {
321 							var initIndex = self.getIndex(ireq.name);
322 							fullproof.call_new_thread(function() {
323 								initIndex.clear(function() {
324 									ireq.initializer(self.getIndex(ireq.name), function() {
325 										fullproof.call_new_thread(callInitializerIfNeeded, database, self, indexRequestArray, callback, errorCallback);
326 										fullproof.call_new_thread(function() {
327 											var tx = database.transaction([self.metaStoreName], fullproof.store.READWRITEMODE);
328 											var metastore = tx.objectStore(self.metaStoreName);
329 											obj.init = true;
330 											install_on_request(metastore.put(obj), function(){}, function(){});;
331 										});
332 									});
333 								});
334 							});
335 						} else {
336 							fullproof.call_new_thread(callInitializerIfNeeded, database, self, indexRequestArray, callback, errorCallback);
337 						}
338 					}, errorCallback);
339 		}

341 		function checkInit(self, database, indexRequestArray, callback, errorCallback) {
342 			createStores(database, indexRequestArray, self.metaStoreName);
343 			setupIndexes(self);
344 			// callInitializerIfNeeded(database, self, [].concat(indexRequestArray), callback, errorCallback);
345             fullproof.call_new_thread(callInitializerIfNeeded, database, self, [].concat(indexRequestArray), callback, errorCallback);
346 		}
347 		
348 		this.indexRequests = reqIndexArray;
349 
350 		var openRequest = fullproof.store.indexedDB.open(this.dbName, this.dbVersion);
351 		openRequest.onerror = function() { 
352 			errorCallback();
353 		};
354 		openRequest.onsuccess = function(ev) {
355 			self.database = ev.result || ev.target.result;
356 					
357 			if (self.database.version !== undefined && self.database.setVersion && self.database.version != self.dbVersion) {
358 				var versionreq = self.database.setVersion(self.dbVersion);
359 				versionreq.onerror = fullproof.make_callback(errorCallback, "Can't change version with setVersion(" +self.dbVersion+")");
360 				versionreq.onsuccess = function(ev) {
361 					createStores(self.database, reqIndexArray, self.metaStoreName);
362 					checkInit(self, self.database, self.indexRequests, 
363 							function() { 
364 								callback(indexArrayResult); 
365 							}, errorCallback);
366 				}
367 			} else {
368 				checkInit(self, self.database, self.indexRequests, fullproof.make_callback(callback, indexArrayResult), errorCallback);
369 			}	
370 		};
371 		openRequest.onupgradeneeded = function(ev) {
372 			createStores(ev.target.result, reqIndexArray, self.metaStoreName);
373 			updated = true;
374 		};
