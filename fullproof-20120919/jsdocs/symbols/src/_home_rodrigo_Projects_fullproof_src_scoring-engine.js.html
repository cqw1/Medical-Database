 19 /**
 20  * This engine is based on scoring. During the injection of document, the parser must provide {fullproof.ScoredElement}
 21  * instances instead of primary values. The score is used to sort the results.
 22  *
 23  * @constructor
 24  */
 25 fullproof.ScoringEngine = function (storeDescriptors) {
 26     if (!(this instanceof fullproof.ScoringEngine)) {
 27         return new fullproof.ScoringEngine(storeDescriptors);
 28     }
 29 
 30     this.initAbstractEngine(storeDescriptors);
 31 };
 32

 34 fullproof.ScoringEngine.prototype = new fullproof.AbstractEngine();

 36 fullproof.ScoringEngine.prototype.checkCapabilities = function (capabilities, analyzer) {
 37     if (capabilities.getUseScores() !== true) {
 38         throw "capabilities.getUseScore() must be true";
 39     }
 40     if (analyzer.provideScore !== true) {
 41         throw "analyzer.provideScore must be true";
 42     }
 43     if (!capabilities.getComparatorObject()) {
 44         throw "capabilities.getComparatorObject() must return a valid comparator";
 45     }
 46 
 47     return true;
 48 };
 49 
 50 fullproof.ScoringEngine.prototype.lookup = function(text, callback) {
 51 	
 52 	var units = this.getIndexUnits();
 53 
 54     function applyScoreModifier(resultset, modifier) {
 55         for (var i= 0, data=resultset.getDataUnsafe(), len=data.length; i<len; ++i){
 56             data[i].score *= modifier;
 57         }
 58     }
 59 
 60 	function merge_resultsets(rset_array, unit) {
 61 		if (rset_array.length == 0) {
 62 			return new fullproof.ResultSet(unit.capabilities.getComparatorObject());
 63 		} else {
 64 			var set = rset_array.shift();
 65 			while (rset_array.length > 0) {
 66 				set.merge(rset_array.shift(), fullproof.ScoredElement.mergeFn);
 67 			}
 68 			return set;
 69 		}
 70 	}
 71 	
 72 	var synchro_all_indexes = fullproof.make_synchro_point(function(array_of_resultset) {
 73 		var merged = merge_resultsets(array_of_resultset);
 74 		callback(merged);
 75 	}, units.length);
 76 	
 77 	for (var i=0; i<units.length; ++i) {
 78 		var unit = units[i];
 79 		unit.analyzer.parse(text, fullproof.make_synchro_point(function(array_of_words) {
 80 			if (array_of_words) {
 81 					if (array_of_words.length == 0) {
 82 						callback(new fullproof.ResultSet(store.caps.getComparatorObject()));
 83 					} else {
 84 						var lookup_synchro = fullproof.make_synchro_point(function(rset_array) {
 85 							var merged = merge_resultsets(rset_array, unit);
 86                             if (unit.capabilities.getScoreModifier() !== undefined) {
 87                                 applyScoreModifier(merged, unit.capabilities.getScoreModifier());
 88                             }
 89 							synchro_all_indexes(merged);
 90 						}, array_of_words.length);
 91 		
 92 						for (var i=0; i<array_of_words.length; ++i) {
 93 							unit.index.lookup(array_of_words[i].key, lookup_synchro);
