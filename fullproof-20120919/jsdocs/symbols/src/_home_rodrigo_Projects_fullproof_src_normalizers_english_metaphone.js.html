 27 	 * Borrowed from https://github.com/NaturalNode/natural/blob/master/lib/natural/phonetics/metaphone.js

 29 	NAMESPACE.english.metaphone = (function(){
 30 		"use strict";
 31 
 32 		function dedup(token) {
 33 		    return token.replace(/([^c])\1/g, '$1');
 34 		}
 35 
 36 		function dropInitialLetters(token) {
 37 		    if(token.match(/^(kn|gn|pn|ae|wr)/))
 38 		        return token.substr(1, token.length - 1);
 39 		        
 40 		    return token;
 41 		}
 42 
 43 		function dropBafterMAtEnd(token) {
 44 		    return token.replace(/mb$/, 'm');
 45 		}
 46 
 47 		function cTransform(token) {
 48 		    token = token.replace(/([^s]|^)(c)(h)/g, '$1x$3').trim();
 49 		    token = token.replace(/cia/g, 'xia');
 50 		    token = token.replace(/c(i|e|y)/g, 's$1');
 51 		    token = token.replace(/c/g, 'k'); 
 52 		    
 53 		    return token;
 54 		}
 55 
 56 		function dTransform(token) {
 57 		    token = token.replace(/d(ge|gy|gi)/g, 'j$1');
 58 		    token = token.replace(/d/g, 't');
 59 		    
 60 		    return token;
 61 		}
 62 
 63 		function dropG(token) {
 64 		    token = token.replace(/gh(^$|[^aeiou])/g, 'h$1');
 65 		    token = token.replace(/g(n|ned)$/g, '$1');    
 66 		    
 67 		    return token;
 68 		}
 69 
 70 		function transformG(token) {
 71 		    token = token.replace(/([^g]|^)(g)(i|e|y)/g, '$1j$3');
 72 		    token = token.replace(/gg/g, 'g');
 73 		    token = token.replace(/g/g, 'k');    
 74 		    
 75 		    return token;
 76 		}
 77 
 78 		function dropH(token) {
 79 		    return token.replace(/([aeiou])h([^aeiou])/g, '$1$2');
 80 		}
 81 
 82 		function transformCK(token) {
 83 		    return token.replace(/ck/g, 'k');
 84 		}
 85 		function transformPH(token) {
 86 		    return token.replace(/ph/g, 'f');
 87 		}
 88 
 89 		function transformQ(token) {
 90 		    return token.replace(/q/g, 'k');
 91 		}
 92 
 93 		function transformS(token) {
 94 		    return token.replace(/s(h|io|ia)/g, 'x$1');
 95 		}
 96 
 97 		function transformT(token) {
 98 		    token = token.replace(/t(ia|io)/g, 'x$1');
 99 		    token = token.replace(/th/, '0');
100 		    
101 		    return token;
102 		}
103 
104 		function dropT(token) {
105 		    return token.replace(/tch/g, 'ch');
106 		}
107 
108 		function transformV(token) {
109 		    return token.replace(/v/g, 'f');
110 		}
111 
112 		function transformWH(token) {
113 		    return token.replace(/^wh/, 'w');
114 		}
115 
116 		function dropW(token) {
117 		    return token.replace(/w([^aeiou]|$)/g, '$1');
118 		}
119 
120 		function transformX(token) {
121 		    token = token.replace(/^x/, 's');
122 		    token = token.replace(/x/g, 'ks');
123 		    return token;
124 		}
125 
126 		function dropY(token) {
127 		    return token.replace(/y([^aeiou]|$)/g, '$1');
128 		}
129 
130 		function transformZ(token) {
131 		    return token.replace(/z/, 's');
132 		}
133 
134 		function dropVowels(token) {
135 		    return token.charAt(0) + token.substr(1, token.length).replace(/[aeiou]/g, '');
136 		}
137 
138 		return function(token, maxLength) {
139 		    maxLength = maxLength || 32;
140 		    token = token.toLowerCase();
141 		    token = dedup(token);
142 		    token = dropInitialLetters(token);
143 		    token = dropBafterMAtEnd(token);
144 		    token = transformCK(token);
145 		    token = cTransform(token);
146 		    token = dTransform(token);
147 		    token = dropG(token);
148 		    token = transformG(token);
149 		    token = dropH(token);
150 		    token = transformPH(token);
151 		    token = transformQ(token);
152 		    token = transformS(token);
153 		    token = transformX(token);    
154 		    token = transformT(token);
155 		    token = dropT(token);
156 		    token = transformV(token);
157 		    token = transformWH(token);
158 		    token = dropW(token);
159 		    token = dropY(token);
160 		    token = transformZ(token);
161 		    token = dropVowels(token);
162 		    
163 		    token.toUpperCase();
164 		    if(token.length >= maxLength)
165 		        token = token.substring(0, maxLength);        
166 
167 		    return token.toUpperCase();
168 		};
169 		
170 	})();
